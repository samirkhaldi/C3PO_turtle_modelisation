select ?personnelCost
where {
:Yann :hoursWorkedOnLumen ?EligibleHoursWorked .
:Yann :gotSalaryGross ?salarygross ;
      :gotPatronalContribution ?patronalcontribution .
  
BIND((?salarygross + ?patronalcontribution) * 12 / 1720 AS ?HourlyRate)
BIND((?EligibleHoursWorked * ?HourlyRate) AS ?personnelCost)
}
