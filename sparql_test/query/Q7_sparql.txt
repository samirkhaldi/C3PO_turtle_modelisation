SELECT ?rpName (SUM(?cost) AS ?totalCost)
WHERE {
  ?rp a :ReportingPeriod ;
      :gotName ?rpName ;
      :startDate ?rpStart ;
      :endDate ?rpEnd .

  # coût des employés
  ?employee :WorkedOnLumenDuring ?rp ;
            :gotSalaryGross ?salary ;
            :gotPatronalContribution ?pc ;
            :hoursWorkedOnLumen ?hours ;
            :gotAnnualProductiveHours ?annualHours .

  BIND ( (?salary + ?pc) * ?hours / ?annualHours AS ?cost )

  # coût des factures
  OPTIONAL {
    ?invoice :duringPeriod ?rp ;
             :amount ?amount .
    BIND (?amount AS ?invoiceCost)
  }

  BIND (COALESCE(?cost, 0) + COALESCE(?invoiceCost, 0) AS ?cost)
}
GROUP BY ?rpName
