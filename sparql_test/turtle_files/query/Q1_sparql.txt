PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> .
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#> .
PREFIX skos:<http://www.w3.org/2004/02/skos/core>.
PREFIX : <https://www.turtle_tutorial.com/> .

SELECT (SUM(?Cost) AS ?TotalPersonnalCostForLumen)
WHERE {
  ?Employee rdf:type :Employee ;
            :gotAnnualProductiveHours ?annualphours ;
            :hoursWorkedOnLumen ?hoursWorked ;
            :gotSalaryGross ?salarygross ;
            :gotPatronalContribution ?patronalcontribution ;
            :WorkedOnLumenDuring :RP3 .
            

  BIND(((?salarygross + ?patronalcontribution) / ?annualphours) * 12 AS ?HourlyRate) .
  BIND((?HourlyRate * ?hoursWorked) AS ?Cost) .
}
