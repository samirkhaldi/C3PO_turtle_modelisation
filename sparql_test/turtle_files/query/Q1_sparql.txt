SELECT (SUM(?cost) AS ?TotalPersonnelCostForLumen)
WHERE {
  ?wr a :WorkRecord ;
      :onProject :LUMEN ;
      :employee ?emp ;
      :hoursWorked ?hours .

  ?emp :gotAnnualProductiveHours ?annualProductiveHours ;
       :gotSalaryGross ?salary ;
       :gotPatronalContribution ?contrib .

  BIND((?salary + ?contrib) * 12 AS ?annualPersonnelCost) 
  BIND((?annualPersonnelCost / ?annualProductiveHours) AS ?hourlyRate)
  BIND(?hourlyRate * ?hours AS ?cost)
}
