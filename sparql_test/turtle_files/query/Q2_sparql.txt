SELECT (SUM((?h / ?annual) * 12) AS ?pm)
WHERE {
  ?wr a :WorkRecord ;
      :employee :Yann ;
      :onProject :LUMEN ;
      :onWorkPackage :WP2 ;
      :duringPeriod :P1 ;
      :hoursWorked ?h .

  :Yann :gotAnnualProductiveHours ?annual .
}
