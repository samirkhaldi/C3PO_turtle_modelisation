SELECT (SUM((?hours / ?annualHours) * 12) AS ?totalPersonMonths)
WHERE {
  :Task1 :includeIn :LUMEN ;
         :RP :P1 ;
         :relatedWP ?wp .

  ?wp rdf:type :WorkPackage ;
      :RP :P1 .

  ?emp rdf:type :Employee ;
       :workOnWP ?wp ;
       :WorkedOnLumenDuring :P1 ;
       :hoursWorkedOnLumen ?hours ;
       :gotAnnualProductiveHours ?annualHours .

  BIND((?hours / ?annualHours) * 12 AS ?personMonth)
}
