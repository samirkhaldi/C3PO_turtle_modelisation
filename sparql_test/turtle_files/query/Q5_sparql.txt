SELECT ?wp (SUM(?pm) AS ?totalPersonMonths)
WHERE {
  ?wr a :WorkRecord ;
      :onProject :LUMEN ;
      :duringPeriod :P1 ;
      :onWorkPackage ?wp ;
      :hoursWorked ?hours .

  BIND((?hours / 1720) * 12 AS ?pm)
}
GROUP BY ?wp
