select ?annualPersonalCost
where { 
:Yann :gotSalaryGross ?salarygross ;
      :gotPatronalContribution ?patronalcontribution .
  
  BIND((?salarygross + ?patronalcontribution) * 12 AS ?annualPersonalCost)
}
