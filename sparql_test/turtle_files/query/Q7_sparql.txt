SELECT ?rpName (?empTotal + ?invTotal AS ?totalCost)
WHERE {

  ?rp a :ReportingPeriod ;
      :gotName ?rpName .

  {
    SELECT ?rp (SUM(?empCost) AS ?empTotal)
    WHERE {
      ?wr a :WorkRecord ;
          :duringPeriod ?rp ;
          :employee ?emp ;
          :hoursWorked ?hours .

      ?emp :gotSalaryGross ?salary ;
           :gotPatronalContribution ?pc ;
           :gotAnnualProductiveHours ?annual .

      BIND(((?salary + ?pc) / ?annual) * ?hours AS ?empCost)
    }
    GROUP BY ?rp
  }

  {
    SELECT ?rp (SUM(?amount) AS ?invTotal)
    WHERE {
      ?invoice :duringPeriod ?rp ;
               :amount ?amount .
    }
    GROUP BY ?rp
  }

}
