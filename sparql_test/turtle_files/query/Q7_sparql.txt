select ?rpName 
       (SUM(?empCost) + SUM(?invCost) AS ?totalCost)
where {
  ?rp a :ReportingPeriod ;
      :gotName ?rpName .

  OPTIONAL {
    ?employee :WorkedOnLumenDuring ?rp ;
              :gotSalaryGross ?salary ;
              :gotPatronalContribution ?pc ;
              :hoursWorkedOnLumen ?hours ;
              :gotAnnualProductiveHours ?annualHours .

    BIND( ((?salary + ?pc) * ?hours) / ?annualHours AS ?empCost )
  }

  OPTIONAL {
    ?invoice :duringPeriod ?rp ;
             :amount ?amount .

    BIND(?amount AS ?invCost)
  }

  BIND(COALESCE(?empCost, 0) AS ?empCost)
  BIND(COALESCE(?invCost, 0) AS ?invCost)
}
GROUP BY ?rpName
